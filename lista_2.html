@dataclass
class OpCounter:
    """Contabiliza operações elementares executadas pelos núcleos numéricos."""

    add: int = 0
    sub: int = 0
    mul: int = 0
    div: int = 0

    def total(self) -> int:
        return self.add + self.sub + self.mul + self.div

    def as_dict(self) -> dict:
        return {
            "soma": self.add,
            "subtracao": self.sub,
            "multiplicacao": self.mul,
            "divisao": self.div,
            "total": self.total(),
        }

    def snapshot(self) -> "OpCounter":
        return OpCounter(self.add, self.sub, self.mul, self.div)

    def __sub__(self, other: "OpCounter") -> "OpCounter":
        return OpCounter(
            self.add - other.add,
            self.sub - other.sub,
            self.mul - other.mul,
            self.div - other.div,
        )


def _bump(counter: Optional[OpCounter], *, add=0, sub=0, mul=0, div=0) -> None:
    if counter is not None:
        counter.add += add
        counter.sub += sub
        counter.mul += mul
        counter.div += div


def k_dot(u: np.ndarray, v: np.ndarray, counter: Optional[OpCounter] = None) -> float:
    """Produto interno u·v de vetores de tamanho m: m mult + (m-1) somas."""
    m = len(u)
    _bump(counter, mul=m, add=max(m - 1, 0))
    return float(np.dot(u, v))


def k_matvec(M: np.ndarray, v: np.ndarray, counter: Optional[OpCounter] = None) -> np.ndarray:
    """Produto matriz-vetor M(p×q)·v: p·q mult + p·(q-1) somas."""
    p, q = M.shape
    _bump(counter, mul=p * q, add=p * max(q - 1, 0))
    return M @ v


def k_vecmat(v: np.ndarray, M: np.ndarray, counter: Optional[OpCounter] = None) -> np.ndarray:
    """Produto vetor-matriz v·M(p×q): q·p mult + q·(p-1) somas (ex.: y = c_B B^{-1})."""
    p, q = M.shape
    _bump(counter, mul=q * p, add=q * max(p - 1, 0))
    return v @ M


def k_pivot(T: np.ndarray, r: int, q: int, counter: Optional[OpCounter] = None) -> None:
    """
    Pivoteamento de Gauss-Jordan no quadro ``T`` (in-place) sobre o elemento (r, q).

    A linha-pivô é normalizada (divisões) e a coluna ``q`` é zerada em todas as
    demais linhas (mult + subtração por elemento). Conta exatamente as operações
    densas executadas.
    """
    rows, cols = T.shape
    p = T[r, q]
    # Normaliza a linha-pivô: divide as (cols-1) entradas != coluna-pivô.
    for k in range(cols):
        if k == q:
            continue
        T[r, k] = T[r, k] / p
        _bump(counter, div=1)
    T[r, q] = 1.0
    # Elimina a coluna q das demais linhas (mult + sub por entrada != coluna-pivô).
    for i in range(rows):
        if i == r:
            continue
        factor = T[i, q]
        for k in range(cols):
            if k == q:
                continue
            T[i, k] = T[i, k] - factor * T[r, k]
            _bump(counter, mul=1, sub=1)
        T[i, q] = 0.0


def k_eta_update(Binv: np.ndarray, d: np.ndarray, r: int,
                 counter: Optional[OpCounter] = None) -> None:
    """
    Atualiza B^{-1} (in-place) por produto elementar (matriz eta) quando a r-ésima
    variável básica sai e entra a variável cuja direção simplex é ``d = B^{-1} A_q``.

    Equivale a pivotar as linhas de B^{-1} usando ``d`` como coluna-pivô:
        nova_linha_r = linha_r / d_r
        nova_linha_i = linha_i - d_i * nova_linha_r      (i != r)
    """
    m = Binv.shape[0]
    dr = d[r]
    for k in range(m):
        Binv[r, k] = Binv[r, k] / dr
        _bump(counter, div=1)
    for i in range(m):
        if i == r:
            continue
        di = d[i]
        for k in range(m):
            Binv[i, k] = Binv[i, k] - di * Binv[r, k]
            _bump(counter, mul=1, sub=1)


@dataclass
class StandardForm:
    A: np.ndarray                # m x N (colunas estruturais + folga/excesso/artificiais)
    b: np.ndarray                # m (>= 0)
    c: np.ndarray                # N — custos da FASE 2 (já em forma de maximização)
    var_names: List[str]         # N nomes de colunas
    basis: List[int]             # m índices de colunas da base inicial
    artificial_idx: List[int]    # índices das colunas artificiais
    is_artificial: List[bool]    # máscara por coluna
    n_structural: int            # número de variáveis de decisão originais
    structural_names: List[str]
    original_sense: str          # 'max' ou 'min'
    obj_sign: int                # +1 (max) ou -1 (min): obj_reportado = obj_sign * obj_max
    senses_norm: List[str]       # sentidos após normalização b >= 0

    @property
    def m(self) -> int:
        return self.A.shape[0]

    @property
    def N(self) -> int:
        return self.A.shape[1]

    @property
    def enterable(self) -> List[bool]:
        """Colunas que PODEM entrar na base (todas, exceto artificiais)."""
        return [not a for a in self.is_artificial]


def to_standard_form(c: Sequence[float], A: Sequence[Sequence[float]],
                     senses: Sequence[str], b: Sequence[float],
                     sense: str = "max",
                     var_names: Optional[Sequence[str]] = None) -> StandardForm:
    """
    Converte (c, A, senses, b, sense) para a forma padrão de igualdades com base
    inicial identidade (folgas para <=, artificiais para >= e =).

    Args:
        c: coeficientes da função objetivo (tamanho n).
        A: matriz de restrições (m x n).
        senses: lista de '<=', '>=' ou '=' (tamanho m).
        b: lado direito (tamanho m).
        sense: 'max' ou 'min'.
        var_names: nomes das variáveis estruturais.

    Returns:
        StandardForm pronto para os solvers de duas fases.
    """
    A = np.array(A, dtype=float)
    b = np.array(b, dtype=float)
    c = np.array(c, dtype=float)
    m, n = A.shape
    if var_names is None:
        var_names = [f"x{j + 1}" for j in range(n)]
    var_names = list(var_names)
    senses = list(senses)

    sense = sense.lower()
    if sense == "min":
        c_max = -c.copy()
        obj_sign = -1
    else:
        c_max = c.copy()
        obj_sign = 1

    A = A.copy()
    b = b.copy()
    # Normaliza b >= 0 (multiplica a linha por -1 e inverte o sentido se b < 0).
    for i in range(m):
        if b[i] < 0:
            A[i, :] *= -1.0
            b[i] *= -1.0
            if senses[i] == "<=":
                senses[i] = ">="
            elif senses[i] == ">=":
                senses[i] = "<="

    cols: List[np.ndarray] = [A[:, j].copy() for j in range(n)]
    names: List[str] = list(var_names)
    cost: List[float] = list(c_max)
    is_art: List[bool] = [False] * n
    basis: List[Optional[int]] = [None] * m
    artificial_idx: List[int] = []

    def add_col(vec: np.ndarray, name: str, cst: float, art: bool) -> int:
        cols.append(vec)
        names.append(name)
        cost.append(cst)
        is_art.append(art)
        return len(cols) - 1

    for i in range(m):
        s = senses[i]
        if s == "<=":
            e = np.zeros(m)
            e[i] = 1.0
            idx = add_col(e, f"s{i + 1}", 0.0, False)   # folga
            basis[i] = idx
        elif s == ">=":
            e = np.zeros(m)
            e[i] = -1.0
            add_col(e, f"e{i + 1}", 0.0, False)         # excesso (surplus)
            a = np.zeros(m)
            a[i] = 1.0
            idx = add_col(a, f"a{i + 1}", 0.0, True)    # artificial
            basis[i] = idx
            artificial_idx.append(idx)
        else:  # '='
            a = np.zeros(m)
            a[i] = 1.0
            idx = add_col(a, f"a{i + 1}", 0.0, True)
            basis[i] = idx
            artificial_idx.append(idx)

    A_std = np.column_stack(cols)
    return StandardForm(
        A=A_std,
        b=b,
        c=np.array(cost, dtype=float),
        var_names=names,
        basis=[int(x) for x in basis],
        artificial_idx=artificial_idx,
        is_artificial=is_art,
        n_structural=n,
        structural_names=var_names,
        original_sense=sense,
        obj_sign=obj_sign,
        senses_norm=senses,
    )


def clean(x: float, tol: float = 1e-9) -> float:
    """Arredonda ruído numérico para zero/inteiro próximo (apenas para exibição)."""
    if abs(x) < tol:
        return 0.0
    r = round(x)
    if abs(x - r) < 1e-7:
        return float(r)
    return float(x)


def df_to_markdown(df, floatfmt: str = "{:.3g}") -> str:
    """Converte um DataFrame em tabela markdown (sem depender de 'tabulate')."""
    cols = list(df.columns)
    header = "| " + " | ".join([str(df.index.name or "")] + [str(c) for c in cols]) + " |"
    sep = "| " + " | ".join(["---"] * (len(cols) + 1)) + " |"
    lines = [header, sep]
    for idx, row in df.iterrows():
        cells = []
        for c in cols:
            v = row[c]
            if isinstance(v, (int, float, np.floating, np.integer)):
                cells.append(floatfmt.format(float(v)))
            else:
                cells.append(str(v))
        lines.append("| " + " | ".join([str(idx)] + cells) + " |")
    return "\n".join(lines)


def fmt_num(x: float, width: int = 9, prec: int = 3) -> str:
    """Formata número para alinhamento em quadros de console."""
    x = clean(x)
    if x == int(x) and abs(x) < 1e6:
        s = f"{int(x)}"
    else:
        s = f"{x:.{prec}f}"
    return s.rjust(width)

@dataclass
class TableauResult:
    status: str                       # 'optimal' | 'unbounded' | 'infeasible'
    objective: Optional[float]        # valor ótimo (sentido original) ou None
    x: Dict[str, float]               # variáveis estruturais -> valor
    multiple_optima: bool = False
    multiple_var: Optional[str] = None
    alt_vertex: Optional[Dict[str, float]] = None
    iterations: List[dict] = field(default_factory=list)
    n_iterations: int = 0
    counter: Optional[OpCounter] = None
    basis_final: List[str] = field(default_factory=list)
    reduced_final: Dict[str, float] = field(default_factory=dict)
    sf: Optional[StandardForm] = None
    entering_unbounded: Optional[str] = None
    phase1_value: Optional[float] = None

    @property
    def solution_type(self) -> str:
        if self.status == "unbounded":
            return "ILIMITADA"
        if self.status == "infeasible":
            return "INFACTÍVEL"
        if self.multiple_optima:
            return "MÚLTIPLAS ÓTIMAS"
        return "ÓTIMA ÚNICA"


def _set_obj_row(T: np.ndarray, basis: List[int], cost: np.ndarray,
                 m: int, N: int) -> None:
    """Recalcula a linha-objetivo r_j = z_j - c_j e Z = c_B·b (setup, não contado)."""
    cB = np.array([cost[b] for b in basis])
    for j in range(N):
        zj = float(np.dot(cB, T[:m, j]))
        T[m, j] = zj - cost[j]
    T[m, N] = float(np.dot(cB, T[:m, N]))


def _snapshot(T: np.ndarray, basis: List[int], sf: StandardForm,
              phase: int, it: int) -> dict:
    m, N = sf.m, sf.N
    index = [sf.var_names[b] for b in basis] + ["Zj - Cj"]
    data = np.vstack([T[:m, :], T[m:m + 1, :]])
    df = pd.DataFrame(
        [[clean(v) for v in data[r, :]] for r in range(m + 1)],
        index=index,
        columns=sf.var_names + ["b"],
    )
    df.index.name = "Base"
    reduced = {sf.var_names[j]: clean(T[m, j]) for j in range(N)}
    return {
        "phase": phase,
        "iter": it,
        "basis": [sf.var_names[b] for b in basis],
        "tableau": df,
        "tableau_raw": T.copy(),
        "reduced": reduced,
        "obj_max": clean(T[m, N]),
        "entering": None,
        "leaving": None,
        "ratios": [],
        "pivot_rc": None,
        "pivot_elem": None,
        "note": "",
        "ops_iter": None,
    }


def _format_ratios(T: np.ndarray, basis: List[int], sf: StandardForm,
                   q: int, m: int, N: int) -> List[dict]:
    out = []
    for i in range(m):
        aiq = T[i, q]
        entry = {
            "base": sf.var_names[basis[i]],
            "b": clean(T[i, N]),
            "col": clean(aiq),
            "ratio": None,
        }
        if aiq > TOL:
            entry["ratio"] = clean(T[i, N] / aiq)
        out.append(entry)
    return out


def _loop(T: np.ndarray, basis: List[int], cost: np.ndarray, sf: StandardForm,
          phase: int, logs: List[dict], counter: Optional[OpCounter],
          max_iter: int) -> str:
    m, N = sf.m, sf.N
    enterable = sf.enterable
    for it in range(max_iter):
        c0 = counter.snapshot() if counter is not None else None
        rec = _snapshot(T, basis, sf, phase, it)
        basic_set = set(basis)

        # Critério de entrada: r_j mais negativo entre não-básicas elegíveis.
        q = None
        best = -TOL
        for j in range(N):
            if not enterable[j] or j in basic_set:
                continue
            if T[m, j] < best:
                best = T[m, j]
                q = j

        if q is None:
            rec["note"] = f"Fase {phase}: ótimo (todos r_j >= 0)."
            rec["ops_iter"] = (counter - c0).as_dict() if counter is not None else None
            logs.append(rec)
            return "optimal"

        rec["entering"] = sf.var_names[q]
        rec["pivot_rc"] = clean(T[m, q])
        rec["ratios"] = _format_ratios(T, basis, sf, q, m, N)

        # Teste da razão mínima.
        leave_r = None
        min_ratio = None
        for i in range(m):
            aiq = T[i, q]
            if aiq > TOL:
                ratio = T[i, N] / aiq
                if counter is not None:
                    counter.div += 1
                if min_ratio is None or ratio < min_ratio - TOL:
                    min_ratio = ratio
                    leave_r = i

        if leave_r is None:
            rec["note"] = (f"Coluna de {sf.var_names[q]} tem todos a_iq <= 0 "
                           f"=> PRIMAL ILIMITADO.")
            rec["ops_iter"] = (counter - c0).as_dict() if counter is not None else None
            logs.append(rec)
            return "unbounded"

        rec["leaving"] = sf.var_names[basis[leave_r]]
        rec["pivot_elem"] = clean(T[leave_r, q])
        rec["note"] = (f"Entra {sf.var_names[q]} (r={rec['pivot_rc']}), "
                       f"sai {rec['leaving']} (razão mín = {clean(min_ratio)}).")
        logs.append(rec)

        k_pivot(T, leave_r, q, counter)
        basis[leave_r] = q
        if counter is not None:
            rec["ops_iter"] = (counter - c0).as_dict()

    raise RuntimeError("Número máximo de iterações excedido (possível ciclagem).")


def _drive_out_artificials(T: np.ndarray, basis: List[int], sf: StandardForm,
                           counter: Optional[OpCounter], logs: List[dict]) -> None:
    """Remove artificiais que permaneceram básicas em nível zero após a Fase 1."""
    m, N = sf.m, sf.N
    art = set(sf.artificial_idx)
    for r in range(m):
        if basis[r] in art and abs(T[r, N]) < 1e-7:
            # Procura coluna não-artificial não-básica com pivô != 0 nesta linha.
            for j in range(N):
                if sf.is_artificial[j] or j in basis:
                    continue
                if abs(T[r, j]) > TOL:
                    k_pivot(T, r, j, counter)
                    basis[r] = j
                    break


def solve_tableau(c, A, senses, b, sense: str = "max",
                  var_names: Optional[List[str]] = None,
                  count_ops: bool = False, max_iter: int = 200) -> TableauResult:
    """
    Resolve o PL pelo Simplex por Quadros (duas fases).

    Args:
        c, A, senses, b: dados do PL (senses em '<=', '>=', '=').
        sense: 'max' ou 'min'.
        var_names: nomes das variáveis estruturais.
        count_ops: se True, contabiliza operações aritméticas dos pivôs.

    Returns:
        TableauResult com status, solução, tipo e log de iterações.
    """
    sf = to_standard_form(c, A, senses, b, sense, var_names)
    m, N = sf.m, sf.N
    counter = OpCounter() if count_ops else None
    logs: List[dict] = []

    T = np.zeros((m + 1, N + 1))
    T[:m, :N] = sf.A
    T[:m, N] = sf.b
    basis = list(sf.basis)

    # ----- FASE 1 (se houver artificiais) -----
    phase1_value = None
    if sf.artificial_idx:
        phase1_cost = np.zeros(N)
        for idx in sf.artificial_idx:
            phase1_cost[idx] = -1.0   # maximiza -(soma das artificiais)
        _set_obj_row(T, basis, phase1_cost, m, N)
        _loop(T, basis, phase1_cost, sf, 1, logs, counter, max_iter)
        phase1_value = clean(T[m, N])           # = -(soma das artificiais)
        if T[m, N] < -1e-7:
            return TableauResult(
                status="infeasible", objective=None, x={},
                iterations=logs, n_iterations=len(logs), counter=counter,
                sf=sf, phase1_value=phase1_value,
            )
        _drive_out_artificials(T, basis, sf, counter, logs)

    # ----- FASE 2 -----
    _set_obj_row(T, basis, sf.c, m, N)
    status = _loop(T, basis, sf.c, sf, 2, logs, counter, max_iter)

    n_iter = sum(1 for L in logs if L["entering"] is not None)

    if status == "unbounded":
        last = logs[-1]
        return TableauResult(
            status="unbounded", objective=None, x={},
            iterations=logs, n_iterations=n_iter, counter=counter, sf=sf,
            entering_unbounded=last["entering"], phase1_value=phase1_value,
        )

    # ----- Ótimo: extrai solução -----
    x_full = np.zeros(N)
    for i in range(m):
        x_full[basis[i]] = T[i, N]
    x = {sf.structural_names[j]: clean(x_full[j]) for j in range(sf.n_structural)}
    obj = sf.obj_sign * T[m, N]
    reduced_final = {sf.var_names[j]: clean(T[m, j]) for j in range(N)}

    # ----- Detecção de múltiplas ótimas -----
    basic_set = set(basis)
    multiple = False
    mult_var = None
    alt_vertex = None
    for j in range(N):
        if sf.enterable[j] and j not in basic_set and abs(T[m, j]) < 1e-7:
            multiple = True
            mult_var = sf.var_names[j]
            # Tenta produzir um segundo vértice ótimo pivotando x_j para dentro.
            leave_r, min_ratio = None, None
            for i in range(m):
                if T[i, j] > TOL:
                    ratio = T[i, N] / T[i, j]
                    if min_ratio is None or ratio < min_ratio - TOL:
                        min_ratio, leave_r = ratio, i
            if leave_r is not None:
                T2 = T.copy()
                basis2 = list(basis)
                k_pivot(T2, leave_r, j, None)
                basis2[leave_r] = j
                xf2 = np.zeros(N)
                for i in range(m):
                    xf2[basis2[i]] = T2[i, N]
                alt_vertex = {sf.structural_names[k]: clean(xf2[k])
                              for k in range(sf.n_structural)}
            break

    return TableauResult(
        status="optimal", objective=clean(obj), x=x,
        multiple_optima=multiple, multiple_var=mult_var, alt_vertex=alt_vertex,
        iterations=logs, n_iterations=n_iter, counter=counter,
        basis_final=[sf.var_names[b] for b in basis],
        reduced_final=reduced_final, sf=sf, phase1_value=phase1_value,
    )


def print_tableau_result(res: TableauResult, title: str = "") -> None:
    """Imprime, formatado, todos os quadros de cada iteração e o desfecho."""
    line = "=" * 78
    if title:
        print(line)
        print(f"  SIMPLEX POR QUADROS — {title}")
        print(line)

    for L in res.iterations:
        tag = f"FASE {L['phase']} · Iteração {L['iter']}"
        print(f"\n--- {tag} ---")
        print(f"Base: {{{', '.join(L['basis'])}}}    Z(max) = {L['obj_max']}")
        print(L["tableau"].to_string(float_format=lambda v: fmt_num(v).strip()))
        if L["entering"]:
            print(f"  -> Entra: {L['entering']}  (r = {L['pivot_rc']})")
            if L["ratios"]:
                rs = []
                for rr in L["ratios"]:
                    r_txt = "—" if rr["ratio"] is None else f"{rr['ratio']:.3g}"
                    rs.append(f"{rr['base']}: {rr['b']:.3g}/{rr['col']:.3g}={r_txt}")
                print("  Razões: " + " | ".join(rs))
        if L["leaving"]:
            print(f"  -> Sai:   {L['leaving']}  (pivô = {L['pivot_elem']})")
        if L["note"]:
            print(f"  {L['note']}")

    print(f"\n{'-' * 40}\nDESFECHO: {res.solution_type}")
    if res.status == "optimal":
        print(f"  Z* = {res.objective}")
        print(f"  x* = {res.x}")
        print(f"  Base ótima: {{{', '.join(res.basis_final)}}}")
        if res.multiple_optima:
            print(f"  Múltiplas ótimas: r_{res.multiple_var} = 0 (não-básica).")
            if res.alt_vertex:
                print(f"  Vértice ótimo alternativo: {res.alt_vertex}")
    elif res.status == "unbounded":
        print(f"  Direção ilimitada via entrada de {res.entering_unbounded}.")
    elif res.status == "infeasible":

@dataclass
class RevisedResult:
    status: str
    objective: Optional[float]
    x: Dict[str, float]
    multiple_optima: bool = False
    multiple_var: Optional[str] = None
    alt_vertex: Optional[Dict[str, float]] = None
    iterations: List[dict] = field(default_factory=list)
    n_iterations: int = 0
    counter: Optional[OpCounter] = None
    basis_final: List[str] = field(default_factory=list)
    reduced_final: Dict[str, float] = field(default_factory=dict)
    sf: Optional[StandardForm] = None
    entering_unbounded: Optional[str] = None
    phase1_value: Optional[float] = None

    @property
    def solution_type(self) -> str:
        if self.status == "unbounded":
            return "ILIMITADA"
        if self.status == "infeasible":
            return "INFACTÍVEL"
        if self.multiple_optima:
            return "MÚLTIPLAS ÓTIMAS"
        return "ÓTIMA ÚNICA"


def _basis_names(basis: List[int], sf: StandardForm) -> List[str]:
    return [sf.var_names[b] for b in basis]


def _alt_vertex_revised(A_std, b, basis, Binv, q, sf) -> Optional[Dict[str, float]]:
    """Calcula um segundo vértice ótimo pivotando x_q (custo reduzido nulo) p/ dentro."""
    m = sf.m
    d = Binv @ A_std[:, q]
    xB = Binv @ b
    leave_r, min_ratio = None, None
    for i in range(m):
        if d[i] > TOL:
            ratio = xB[i] / d[i]
            if min_ratio is None or ratio < min_ratio - TOL:
                min_ratio, leave_r = ratio, i
    if leave_r is None:
        return None
    Binv2 = Binv.copy()
    k_eta_update(Binv2, d, leave_r, None)
    basis2 = list(basis)
    basis2[leave_r] = q
    xB2 = Binv2 @ b
    xf = np.zeros(sf.N)
    for i in range(m):
        xf[basis2[i]] = xB2[i]
    return {sf.structural_names[k]: clean(xf[k]) for k in range(sf.n_structural)}


def _rev_loop(A_std, b, cost, basis, Binv, sf, phase, logs, counter, max_iter) -> str:
    m, N = sf.m, sf.N
    enterable = sf.enterable
    for it in range(max_iter):
        c0 = counter.snapshot() if counter is not None else None
        cB = np.array([cost[bi] for bi in basis])
        xB = k_matvec(Binv, b, counter)          # B^{-1} b
        y = k_vecmat(cB, Binv, counter)          # c_B B^{-1}  (multiplicadores)

        basic_set = set(basis)
        reduced: Dict[str, float] = {}
        q, best = None, -TOL
        for j in range(N):
            if j in basic_set or not enterable[j]:
                continue
            zj = k_dot(y, A_std[:, j], counter)
            rj = zj - cost[j]
            if counter is not None:
                counter.sub += 1
            reduced[sf.var_names[j]] = clean(rj)
            if rj < best:
                best, q = rj, j

        rec = {
            "phase": phase, "iter": it,
            "basis": _basis_names(basis, sf),
            "Binv": pd.DataFrame(
                [[clean(v) for v in Binv[r]] for r in range(m)],
                index=_basis_names(basis, sf),
                columns=[f"e{c+1}" for c in range(m)]),
            "cB": {sf.var_names[basis[i]]: clean(cB[i]) for i in range(m)},
            "y": [clean(v) for v in y],
            "xB": {sf.var_names[basis[i]]: clean(xB[i]) for i in range(m)},
            "reduced": reduced,
            "entering": None, "direction": None, "ratios": [],
            "leaving": None, "pivot_rc": None, "note": "", "ops_iter": None,
        }

        if q is None:
            rec["note"] = f"Fase {phase}: ótimo (todos r_j >= 0)."
            rec["ops_iter"] = (counter - c0).as_dict() if counter is not None else None
            logs.append(rec)
            return "optimal"

        d = k_matvec(Binv, A_std[:, q], counter)  # direção simplex
        rec["entering"] = sf.var_names[q]
        rec["pivot_rc"] = clean(best)
        rec["direction"] = {sf.var_names[basis[i]]: clean(d[i]) for i in range(m)}

        leave_r, min_ratio = None, None
        ratios = []
        for i in range(m):
            if d[i] > TOL:
                ratio = xB[i] / d[i]
                if counter is not None:
                    counter.div += 1
                ratios.append({"base": sf.var_names[basis[i]],
                               "xB": clean(xB[i]), "d": clean(d[i]),
                               "ratio": clean(ratio)})
                if min_ratio is None or ratio < min_ratio - TOL:
                    min_ratio, leave_r = ratio, i
            else:
                ratios.append({"base": sf.var_names[basis[i]],
                               "xB": clean(xB[i]), "d": clean(d[i]), "ratio": None})
        rec["ratios"] = ratios

        if leave_r is None:
            rec["note"] = (f"Direção d = B^-1 A_{sf.var_names[q]} <= 0 "
                           f"=> PRIMAL ILIMITADO.")
            rec["ops_iter"] = (counter - c0).as_dict() if counter is not None else None
            logs.append(rec)
            return "unbounded"

        rec["leaving"] = sf.var_names[basis[leave_r]]
        rec["note"] = (f"Entra {sf.var_names[q]} (r={rec['pivot_rc']}), "
                       f"sai {rec['leaving']} (razão mín = {clean(min_ratio)}).")
        logs.append(rec)

        k_eta_update(Binv, d, leave_r, counter)
        basis[leave_r] = q
        if counter is not None:
            rec["ops_iter"] = (counter - c0).as_dict()

    raise RuntimeError("Número máximo de iterações excedido (possível ciclagem).")


def _drive_out_artificials_rev(A_std, b, basis, Binv, sf, counter) -> None:
    m, N = sf.m, sf.N
    art = set(sf.artificial_idx)
    xB = Binv @ b
    for r in range(m):
        if basis[r] in art and abs(xB[r]) < 1e-7:
            for j in range(N):
                if sf.is_artificial[j] or j in basis:
                    continue
                dj = Binv @ A_std[:, j]
                if abs(dj[r]) > TOL:
                    k_eta_update(Binv, dj, r, counter)
                    basis[r] = j
                    break


def solve_revised(c, A, senses, b, sense: str = "max",
                  var_names: Optional[List[str]] = None,
                  count_ops: bool = False, max_iter: int = 200) -> RevisedResult:
    """Resolve o PL pelo Simplex Revisado (duas fases, B^{-1} explícita)."""
    sf = to_standard_form(c, A, senses, b, sense, var_names)
    m, N = sf.m, sf.N
    counter = OpCounter() if count_ops else None
    logs: List[dict] = []

    A_std = sf.A
    bvec = sf.b
    basis = list(sf.basis)
    Binv = np.eye(m)

    phase1_value = None
    if sf.artificial_idx:
        cost1 = np.zeros(N)
        for idx in sf.artificial_idx:
            cost1[idx] = -1.0
        _rev_loop(A_std, bvec, cost1, basis, Binv, sf, 1, logs, counter, max_iter)
        cB = np.array([cost1[bi] for bi in basis])
        w = float(np.dot(cB, Binv @ bvec))
        phase1_value = clean(w)
        if w < -1e-7:
            return RevisedResult(
                status="infeasible", objective=None, x={},
                iterations=logs, n_iterations=len(logs), counter=counter,
                sf=sf, phase1_value=phase1_value)
        _drive_out_artificials_rev(A_std, bvec, basis, Binv, sf, counter)

    status = _rev_loop(A_std, bvec, sf.c, basis, Binv, sf, 2, logs, counter, max_iter)
    n_iter = sum(1 for L in logs if L["entering"] is not None)

    if status == "unbounded":
        return RevisedResult(
            status="unbounded", objective=None, x={},
            iterations=logs, n_iterations=n_iter, counter=counter, sf=sf,
            entering_unbounded=logs[-1]["entering"], phase1_value=phase1_value)

    xB = Binv @ bvec
    x_full = np.zeros(N)
    for i in range(m):
        x_full[basis[i]] = xB[i]
    x = {sf.structural_names[j]: clean(x_full[j]) for j in range(sf.n_structural)}
    cB = np.array([sf.c[bi] for bi in basis])
    obj = sf.obj_sign * float(np.dot(cB, xB))

    # custos reduzidos finais (todas as não-básicas elegíveis)
    y = cB @ Binv
    reduced_final = {}
    basic_set = set(basis)
    multiple, mult_var, alt_vertex = False, None, None
    for j in range(N):
        if j in basic_set or not sf.enterable[j]:
            continue
        rj = float(np.dot(y, A_std[:, j])) - sf.c[j]
        reduced_final[sf.var_names[j]] = clean(rj)
        if abs(rj) < 1e-7 and not multiple:
            multiple, mult_var = True, sf.var_names[j]
            alt_vertex = _alt_vertex_revised(A_std, bvec, basis, Binv, j, sf)

    return RevisedResult(
        status="optimal", objective=clean(obj), x=x,
        multiple_optima=multiple, multiple_var=mult_var, alt_vertex=alt_vertex,
        iterations=logs, n_iterations=n_iter, counter=counter,
        basis_final=_basis_names(basis, sf), reduced_final=reduced_final,
        sf=sf, phase1_value=phase1_value)

from typing import Dict, List, Optional, Sequence

import gurobipy as gp
from gurobipy import GRB


def verify_lp(c: Sequence[float], A: Sequence[Sequence[float]],
              senses: Sequence[str], b: Sequence[float],
              sense: str = "max", var_names: Optional[List[str]] = None,
              save_lp: Optional[str] = None) -> Dict:
    """
    Resolve o PL com o Gurobi e devolve status/objetivo/solução.

    Args:
        c, A, senses, b: dados do PL (senses em '<=', '>=', '=').
        sense: 'max' ou 'min'.
        var_names: nomes das variáveis estruturais.
        save_lp: se fornecido, grava o modelo em formato .lp nesse caminho.

    Returns:
        dict com chaves: status ('optimal'|'unbounded'|'infeasible'),
        objective (ou None), x (dict ou None), gurobi_status_code.
    """
    n = len(c)
    if var_names is None:
        var_names = [f"x{j + 1}" for j in range(n)]

    model = gp.Model()
    model.Params.OutputFlag = 0
    model.Params.DualReductions = 0   # distingue UNBOUNDED de INFEASIBLE
    x = [model.addVar(lb=0.0, name=var_names[j]) for j in range(n)]
    model.setObjective(
        gp.quicksum(c[j] * x[j] for j in range(n)),
        GRB.MAXIMIZE if sense.lower() == "max" else GRB.MINIMIZE,
    )
    for i in range(len(b)):
        lhs = gp.quicksum(A[i][j] * x[j] for j in range(n))
        if senses[i] == "<=":
            model.addConstr(lhs <= b[i], name=f"c{i + 1}")
        elif senses[i] == ">=":
            model.addConstr(lhs >= b[i], name=f"c{i + 1}")
        else:
            model.addConstr(lhs == b[i], name=f"c{i + 1}")

    if save_lp:
        import os
        os.makedirs(os.path.dirname(save_lp), exist_ok=True)
        model.write(save_lp)

    model.optimize()
    st = model.Status
    state_map = {
        GRB.OPTIMAL: "optimal",
        GRB.INFEASIBLE: "infeasible",
        GRB.UNBOUNDED: "unbounded",
        GRB.INF_OR_UNBD: "inf_or_unbounded",
    }
    out: Dict = {"status": state_map.get(st, str(st)), "gurobi_status_code": st}
    if st == GRB.OPTIMAL:
        out["objective"] = model.ObjVal
        out["x"] = {v.VarName: v.X for v in model.getVars()}
    else:
        out["objective"] = None
        out["x"] = None
    return out


def gb(c, A, senses, b, sense="max", vn=None):
    g = verify_lp(c, A, senses, b, sense, vn)
    return g["status"], g["objective"]

Item $\alpha$ $\beta$ $\varphi$ $\delta$ $\mu$ $\varepsilon$

c) Multiplas $\ge 0$ $\ge 0$ irrestrita irrestrita $= 0$ $> 0$
d) Inviavel $< 0$ * $< 0$ * irrestrita irrestrita irrestrita irrestrita

def quadro(alpha, beta, phi, delta, mu, eps):
    return np.array([[-2, 1, delta, 0, 0, beta],
                     [-4, 0, phi,   1, 0, 4],
                     [-2, 0, -1,    0, 1, alpha],
                     [eps, 0, mu,   0, 0, 10]], float)

def classifica(T, tol=1e-9):
    rhs, r = T[:3, 5], T[3, :5]
    if np.any(rhs < -tol):
        return 'inviavel'
    melhora = [j for j in NB if r[j] < -tol]
    for j in melhora:
        if np.all(T[:3, j] <= tol):
            return 'ilimitada'
    if melhora:
        return 'nao-otima'
    return 'multiplas' if any(abs(r[j]) <= tol for j in NB) else 'otima-unica'

Exercicio 4 - Pseudocodigo e Fluxograma (Simplex Revisado)¶

Pseudocodigo (duas fases, $B^{-1}$ explicita):

  1. Padronizar (folga/excesso/artificial); $B^{-1}=I$; base inicial.
  2. Fase 1 (se ha artificiais): max $-\sum a_i$. Se $\sum a_i>0 \Rightarrow$ INFACTIVEL.
  3. Fase 2 — repetir:
    • $x_B = B^{-1}b$ ; $y = c_B B^{-1}$
    • custos reduzidos $r_j = y A_j - c_j$ (nao-basicas)
    • se todo $r_j \ge 0 \Rightarrow$ OTIMO (UNICA se todos $>0$; MULTIPLA se algum $=0$)
    • senao entra $q=\arg\min_j r_j$
    • direcao $d = B^{-1}A_q$ ; se $d \le 0 \Rightarrow$ ILIMITADA
    • razao minima $t^*=\min_{d_i>0}(x_B)_i/d_i \Rightarrow$ sai $r$
    • atualiza $B^{-1} \leftarrow E B^{-1}$ (matriz eta) ; $base[r]=q$
